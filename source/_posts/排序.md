# 排序算法

## 快速排序

```java
public class Main {
    static void quickSort(int[] arr, int start, int end) {
        if (start > end) return;
        int t = arr[start];
        int i = start, j = end;
        while (i < j) {
            while (i < j && arr[j] >= t) j--;
            if (i < j) {
                arr[i] = arr[j];
                i++;
            }
            while (i < j && arr[i] < t) i++;
            if (i < j) {
                arr[j] = arr[i];
                j--;
            }
        }
        arr[i] = t;
        System.out.println(Arrays.toString(arr));
        quickSort(arr, start, i - 1);
        quickSort(arr, j + 1, end);
    }

    public static void main(String[] args) {
        int[] arr = new int[]{4, 7, 2, 3, 6, 5, 1};
        quickSort(arr, 0, arr.length - 1);
        System.out.println(Arrays.toString(arr));
    }
}
```
